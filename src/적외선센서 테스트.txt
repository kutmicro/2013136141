
 
int T = 2; // INPUT PIN
 
// 프로그램 시작 - 초기화 작업
void setup()
{
 Serial.begin(115200); // 시리얼 통신 초기화
 Serial.println("Arduino Examples - IR Sensor");
 Serial.println(" http://docs.whiteat.com/?p=1142");
 pinMode(T, INPUT);
}
 
void loop()
{
 if (LOW == digitalRead(T))
 {
 Serial.println("IR Detected"); // 감지됨
 }
 else
 {
 Serial.print("."); 
 }
 delay(100);
}
