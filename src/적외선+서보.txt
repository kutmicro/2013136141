적외선센서 + 서보모터 ( 장애물 감지  -> 서보모터 정지
			장애물 감지x -> 서보모터 동작 )



#include <AFMotor.h>
#include <Servo.h> 
#include <DHT.h>

// DC motor on M2
AF_DCMotor motor(2);
// DC hobby servo
Servo servo1;
// Stepper motor on M3+M4 48 steps per revolution
AF_Stepper stepper(48, 2);

const int irDetectPin = 22; //FC-51 Pin 설정
const int ledPin = 23;      //LED Pin 설정

void setup() {
  pinMode(irDetectPin, INPUT); //FC-51 입력으로 설정
  pinMode(ledPin, OUTPUT);     //LED 출력으로 설정

  Serial.begin(9600);           // set up Serial library at 9600 bps
  Serial.println("Motor party!");
  
  // turn on servo
  servo1.attach(9);
   
  // turn on motor #2
  motor.setSpeed(200);
  motor.run(RELEASE);
}

int i;

void loop() {
  int noDetect = digitalRead(irDetectPin); //FC-51 상태 읽어오기
  if(noDetect) //FC-51포트가 HIGH이면 즉, 감지 안되었다면
    {
      digitalWrite(ledPin, 0); //LED Off
     motor.run(FORWARD);
  for (i=0; i<255; i++) {
    servo1.write(i);
    motor.setSpeed(i);  
    stepper.step(1, FORWARD, INTERLEAVE);
    delay(1);
 }
 
  for (i=255; i!=0; i--) {
    servo1.write(i-255);
    motor.setSpeed(i);  
    stepper.step(1, BACKWARD, INTERLEAVE);
    delay(1);
 }
 
  motor.run(BACKWARD);
  for (i=0; i<255; i++) {
    servo1.write(i);
    motor.setSpeed(i);  
    delay(1);
    stepper.step(1, FORWARD, DOUBLE);
 }
 
  for (i=255; i!=0; i--) {
    servo1.write(i-255);
    motor.setSpeed(i);  
    stepper.step(1, BACKWARD, DOUBLE);
    delay(1);
 }
    }
  else         //LOW 즉, 감지 되었다면
  {
    digitalWrite(ledPin, 1); //LED ON
      motor.setSpeed(0);
  }
  
 
}
